/**
 * POST /api/admin/rooms/upload-image
 * Upload a room image (Admin only)
 * Body: FormData with 'image' file
 * Returns: {imagePath: "/uploads/room-images/filename.jpg"}
 */
exports.uploadRoomImage = async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: 'No image file provided' });
    }

    // Generate the image path that will be served
    const imagePath = `/uploads/room-images/${req.file.filename}`;

    res.json({
      success: true,
      imagePath: imagePath,
      message: 'Image uploaded successfully'
    });
  } catch (error) {
    console.error('Error uploading image:', error);
    res.status(500).json({ error: error.message });
  }
};
